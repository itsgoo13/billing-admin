function generateInvoice(nama, ip, paket, harga, diskon, tanggal = null, telepon = '', keterangan = '') { const total = parseInt(harga) - parseInt(diskon || 0); const invoiceNumber = 'INV-' + Utilities.formatDate(new Date(), "GMT+7", "yyyyMMdd") + '-' + Math.floor(Math.random()*1000); // Gunakan tanggal yang dikirim atau hari ini let tglInvoice = tanggal ? new Date(tanggal) : new Date(); const tglStr = Utilities.formatDate(tglInvoice, "GMT+7", "dd MMMM yyyy"); const html = `
🌐 R31NET INTERNET SERVICE
Jl Otista Gg H Maung Rt 09/018 No 31, Kedaung - Pamulang
☎ ${telepon || '0878-8986-3799'}
Invoice #: ${invoiceNumber}
Tanggal: ${tglStr}
Pelanggan: ${nama}
IP: ${ip}
${diskon > 0 ? `` : ''}
DeskripsiJumlah
Paket ${paket}Rp ${harga.toLocaleString('id')}
Diskon- Rp ${diskon.toLocaleString('id')}
TOTAL: Rp ${total.toLocaleString('id')}
${keterangan ? `
📝 Catatan: ${keterangan}
` : ''}
Pembayaran:
BCA: 7310685440 a.n VIA DESIANA
DANA/OVO: 087898863799

Terima kasih atas kepercayaan Anda!

`; const blob = Utilities.newBlob(html, 'text/html'); const pdf = blob.getAs('application/pdf').setName('Invoice_' + nama + '.pdf'); let fileUrl = ''; try { // Ganti 'YOUR_FOLDER_ID' dengan ID folder Drive (opsional) const folder = DriveApp.getFolderById('YOUR_FOLDER_ID'); const file = folder.createFile(pdf); fileUrl = file.getUrl(); } catch (e) { const tempFile = DriveApp.createFile(pdf); fileUrl = tempFile.getUrl(); } return { success: true, url: fileUrl }; }
