function generateInvoice(nama, ip, paket, harga, diskon, instalasi = 0, sewaAlat = 0, tanggal = null, telepon = '', keterangan = '') { const hargaNum = parseInt(harga) || 0; const diskonNum = parseInt(diskon) || 0; const instalasiNum = parseInt(instalasi) || 0; const sewaNum = parseInt(sewaAlat) || 0; const subtotal = hargaNum + instalasiNum + sewaNum; const total = subtotal - diskonNum; const invoiceNumber = 'INV-' + Utilities.formatDate(new Date(), "GMT+7", "yyyyMMdd") + '-' + Math.floor(Math.random() * 1000); const tglInvoice = tanggal ? new Date(tanggal) : new Date(); const tglStr = Utilities.formatDate(tglInvoice, "GMT+7", "dd/MM/yyyy"); // Informasi kontak default jika telepon kosong const noTelp = telepon || '0878-8986-3799'; const html = `
R31NET
Internet Service Provider
Jl. Otista Gg. H Maung, Kedaung - Pamulang
☎ ${noTelp}
No. Invoice: ${invoiceNumber}
Tanggal: ${tglStr}
Pelanggan: ${nama}
IP Address: ${ip}
${instalasiNum > 0 ? `` : ''} ${sewaNum > 0 ? `` : ''} ${diskonNum > 0 ? `` : ''}
Paket ${paket}Rp ${hargaNum.toLocaleString('id')}
Biaya InstalasiRp ${instalasiNum.toLocaleString('id')}
Sewa AlatRp ${sewaNum.toLocaleString('id')}
Diskon- Rp ${diskonNum.toLocaleString('id')}
TOTAL: Rp ${total.toLocaleString('id')}
${keterangan ? `
📝 ${keterangan}
` : ''}
Metode Pembayaran:
• Transfer Bank BCA: 7310685440 a.n VIA DESIANA
• DANA / OVO: 087898863799
• Scan QRIS (terlampir di bawah)
Scan untuk pembayaran
--- Terima kasih atas kepercayaan Anda ---
`; const blob = Utilities.newBlob(html, 'text/html'); const pdf = blob.getAs('application/pdf').setName('Invoice_' + nama + '.pdf'); let fileUrl = ''; try { // Ganti 'YOUR_FOLDER_ID' dengan ID folder Drive (opsional) const folder = DriveApp.getFolderById('YOUR_FOLDER_ID'); const file = folder.createFile(pdf); fileUrl = file.getUrl(); } catch (e) { const tempFile = DriveApp.createFile(pdf); fileUrl = tempFile.getUrl(); } return { success: true, url: fileUrl }; }
